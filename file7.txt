def safe_auc_with_reason(y_vals, p_vals):
    df_tmp = pd.DataFrame(
        {
            "y": pd.Series(y_vals),
            "p": pd.Series(p_vals)

    ).dropna()

    if int(df_tmp.shape[0]) == 0:
        return np.nan, "no_rows"
    if int(df_tmp["y"].nunique()) < 2:
        return np.nan, "single_class"
    return float(roc_auc_score(df_tmp["y"], df_tmp["p"])), "ok"

def attach_thin_flag(x_df, source_df):
    if source_df is None:
        return pd.Series(np.full(len(x_df), np.nan), index=x_df.index)

    src_idx = source_df.index
    x_idx = x_df.index
    overlap = len(pd.Index(x_idx).intersection(pd.Index(src_idx)))
    overlap_rate = float(overlap) / float(len(x_df)) if len(x_df) > 0 else 0.0

    if overlap_rate > 0.95:
        return source_df.reindex(x_df.index)["thin_file_flag"]

    candidate_keys = [
        "app_id","application_id","loan_id","account_id","acct_id",
        "customer_id","member_id","user_id","id"
    ]
    usable_keys = [k for k in candidate_keys if (k in x_df.columns) and (k in source_df.columns)]
    if len(usable_keys) > 0:
        join_key = usable_keys[0]
        src_map = (
            source_df[[join_key, "thin_file_flag"]]
            .drop_duplicates(subset=[join_key])
            .set_index(join_key)["thin_file_flag"]
        )
        return x_df[join_key].map(src_map)

    return pd.Series(np.full(len(x_df), np.nan), index=x_df.index)

def binarize_preserve_nan(ser):
    ser_num = pd.to_numeric(ser, errors="coerce")
    out_vals = np.where(pd.isna(ser_num.values), np.nan, (ser_num.values >= 0.5).astype(int))
    return pd.Series(out_vals, index=ser.index)

def flag_is_informative(flag_vals, min_pos=50, min_neg=50):
    ser = pd.to_numeric(pd.Series(flag_vals), errors="coerce")
    miss_rate = float(ser.isna().mean())
    vc = ser.value_counts(dropna=True).to_dict()
    n0 = int(vc.get(0.0, 0))
    n1 = int(vc.get(1.0, 0))
    informative = (n0 >= min_neg) and (n1 >= min_pos)
    diag = {"n0": n0, "n1": n1, "missing_rate": miss_rate}
    return informative, diag

# --- Deciles helper ---
def build_decile_table(tmp_df, score_col, label_prefix):
    tmp_df2 = tmp_df.copy()
    tmp_df2 = tmp_df2[pd.notna(tmp_df2[score_col])]

    if int(len(tmp_df2)) == 0:
        note_df = pd.DataFrame({"note": [label_prefix + ": no non-null scores to form deciles"], "n": [0]})
        print(note_df)
        return

    uniq_scores = int(tmp_df2[score_col].nunique(dropna=True))
    if int(uniq_scores) < 10:
        note_df = pd.DataFrame({
            "note": [label_prefix + ": not enough unique scores for deciles"],
            "unique_scores": [uniq_scores],
            "n": [len(tmp_df2)]
        })
        print(note_df)

        q_vals = tmp_df2[score_col].quantile([0, 0.1, 0.5, 0.9, 1.0]).reset_index()
        q_vals.columns = ["quantile", score_col]
        print(q_vals)
        return

    tmp_df2["decile"] = pd.qcut(tmp_df2[score_col], 10, labels=False, duplicates="drop")
    tmp_df2["decile"] = tmp_df2["decile"].astype(int) + 1

    dec_tbl = (
        tmp_df2.groupby("decile")
        .agg(
            n=("bad_state", "size"),
            bad_rate=("bad_state", "mean"),
            avg_score=(score_col, "mean"),
            min_score=(score_col, "min"),
            max_score=(score_col, "max")
        )
        .reset_index()
        .sort_values("decile")
    )
    print(dec_tbl)
