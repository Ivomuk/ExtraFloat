def build_decile_table_return(tmp_df, score_col, label_prefix, n_bins=10, target_col="bad_state"):

    uniq_scores2 = int(tmp_df2[score_col].nunique(dropna=True))
    if uniq_scores2 < int(n_bins):
        return pd.DataFrame({
            "note": [label_prefix + ": not enough unique scores for deciles"],
            "unique_scores": [uniq_scores2],
            "n": [int(tmp_df2.shape[0])]
        })      tmp_df2["decile"] = pd.qcut(tmp_df2[score_col], int(n_bins), labels=False, duplicates="drop").astype(int) + 1
    dec_tbl2 = tmp_df2.groupby("decile").agg(
        n=(target_col, "size"),
        bad_rate=(target_col, "mean"),
        avg_score=(score_col, "mean"),
        min_score=(score_col, "min"),
        max_score=(score_col, "max")
    ).reset_index().sort_values("decile")

def build_decile_tables(
    df_in,
    score_col="raw_score",
    target_col="bad_state",
    segment_col=None,
    n_bins=10,
    thin_flag_col="thin_file_flag",
    make_thin_segment_if_missing=True,
    thin_threshold=0.5
):
    df_work = df_in.copy()

    # Coerce score/target
    df_work[score_col] = pd.to_numeric(df_work[score_col], errors="coerce")
    df_work[target_col] = pd.to_numeric(df_work[target_col], errors="coerce").fillna(0).astype(int)

    # Segment selection
    if segment_col is not None and segment_col in df_work.columns:
        seg_col_use = segment_col
        df_work[seg_col_use] = df_work[seg_col_use].astype("object").fillna("missing")
    else:
        if make_thin_segment_if_missing and thin_flag_col in df_work.columns:
            thin_series = df_work[thin_flag_col]
            if thin_series.dtype == bool:
                thin_num = thin_series.astype(int)
            else:
                thin_num = pd.to_numeric(thin_series, errors="coerce")

            thin_bin = np.where(pd.isna(thin_num), np.nan, (thin_num >= float(thin_threshold)).astype(int))
            df_work["thin_segment"] = pd.Series(thin_bin, index=df_work.index).map({0: "non_thin", 1: "thin"}).fillna("missing_flag")
            seg_col_use = "thin_segment"
        else:
            df_work["__segment__"] = "all"
            seg_col_use = "__segment__"

    # Segment summary
    seg_vals = pd.Series(df_work[seg_col_use].unique()).dropna().tolist()
    seg_rows = []
    for seg_name in seg_vals:
        seg_df = df_work[df_work[seg_col_use] == seg_name].copy()
        if int(seg_df.shape[0]) == 0:
            continue

        y_vals = seg_df[target_col].values
        p_vals = seg_df[score_col].values
        mask = ~pd.isna(p_vals)

        auc_val = np.nan
        if int(np.sum(mask)) > 0 and int(np.unique(y_vals[mask]).shape[0]) >= 2:
            try:
                auc_val = roc_auc_score(y_vals[mask], p_vals[mask])
            except Exception:
                auc_val = np.nan

        seg_rows.append({
            "segment": seg_name,
            "n": int(seg_df.shape[0]),
            "bad_rate": float(np.mean(y_vals)) if int(seg_df.shape[0]) > 0 else np.nan,
            "avg_score": float(np.nanmean(p_vals)) if int(seg_df.shape[0]) > 0 else np.nan,
            "auc": auc_val,
            "score_null_rate": float(np.mean(pd.isna(p_vals)))
        })

    seg_summary_df = pd.DataFrame(seg_rows)

    # Stable ordering for thin segmentation
    if seg_col_use == "thin_segment" and int(seg_summary_df.shape[0]) > 0:
        seg_order = pd.Categorical(seg_summary_df["segment"], categories=["non_thin", "thin", "missing_flag"], ordered=True)
        seg_summary_df = seg_summary_df.assign(segment=seg_order).sort_values(["segment"]).reset_index(drop=True)
    else:
        seg_summary_df = seg_summary_df.sort_values(["segment"]).reset_index(drop=True) if int(seg_summary_df.shape[0]) > 0 else seg_summary_df

    # Decile helper
    def _deciles_for(sub_df, label):
        tmp_df = sub_df.copy()
        tmp_df = tmp_df[pd.notna(tmp_df[score_col])]
        if int(tmp_df.shape[0]) == 0:
            return pd.DataFrame({"note": [label + ": no non-null scores to form deciles"], "n": [0]})

        uniq_scores = int(tmp_df[score_col].nunique(dropna=True))
        if uniq_scores < int(n_bins):
            return pd.DataFrame({
                "note": [label + ": not enough unique scores for deciles"],
                "unique_scores": [uniq_scores],
                "n": [int(tmp_df.shape[0])]
            })

        tmp_df["decile"] = pd.qcut(tmp_df[score_col], int(n_bins), labels=False, duplicates="drop").astype(int) + 1
        dec_tbl = tmp_df.groupby("decile").agg(
            n=(target_col, "size"),
            bad_rate=(target_col, "mean"),
            avg_score=(score_col, "mean"),
            min_score=(score_col, "min"),
            max_score=(score_col, "max")
        ).reset_index().sort_values("decile")

        overall_bad = float(tmp_df[target_col].mean())
        dec_tbl["lift_vs_overall"] = dec_tbl["bad_rate"] / (overall_bad + 1e-12)
        return dec_tbl       overall_deciles_df = _deciles_for(df_work, "overall")

    deciles_by_segment = {}
    for seg_name in seg_vals:
        seg_df = df_work[df_work[seg_col_use] == seg_name].copy()
        deciles_by_segment[str(seg_name)] = _deciles_for(seg_df, "segment=" + str(seg_name))

    return {
        "seg_col": seg_col_use,
        "seg_summary": seg_summary_df,
        "overall": overall_deciles_df,
        "by_segment": deciles_by_segment
    }


def compare_models_deciles(
    df_a,
    df_b,
    score_a="raw_score",
    score_b="raw_score",
    target_col="bad_state",
    segment_col="thin_segment",
    n_bins=10,
    name_a="model_a",
    name_b="model_b",
    allow_auto_thin_segment=True
):
    # Build deciles for each model. If the requested segment_col isn't present,
    # build_decile_tables will auto-create thin_segment from thin_file_flag.
    res_a = build_decile_tables(
        df_a,
        score_col=score_a,
        target_col=target_col,
        segment_col=segment_col if (segment_col in df_a.columns) else None,
        n_bins=n_bins,
        make_thin_segment_if_missing=allow_auto_thin_segment
    )
    res_b = build_decile_tables(
        df_b,
        score_col=score_b,
        target_col=target_col,
        segment_col=segment_col if (segment_col in df_b.columns) else None,
        n_bins=n_bins,
        make_thin_segment_if_missing=allow_auto_thin_segment
    )

    # Overall AUC on all rows
    def _auc_overall(df_in, score_col_use):
        df_tmp = df_in.copy()
        df_tmp[score_col_use] = pd.to_numeric(df_tmp[score_col_use], errors="coerce")
        df_tmp[target_col] = pd.to_numeric(df_tmp[target_col], errors="coerce").fillna(0).astype(int)
        mask = ~pd.isna(df_tmp[score_col_use])
        if int(np.sum(mask)) == 0:
            return np.nan
        if int(np.unique(df_tmp.loc[mask, target_col].values).shape[0]) < 2:
            return np.nan
        return roc_auc_score(df_tmp.loc[mask, target_col].values, df_tmp.loc[mask, score_col_use].values)

    auc_a = _auc_overall(df_a, score_a)
    auc_b = _auc_overall(df_b, score_b)

    # Top/bottom decile bad rates from overall decile table
    def _top_decile_bad(dec_df):
        if "decile" not in dec_df.columns:
            return np.nan
        top_dec = int(dec_df["decile"].max())
        vals = dec_df.loc[dec_df["decile"] == top_dec, "bad_rate"].values
        return float(vals[0]) if int(vals.shape[0]) > 0 else np.nan

    def _bottom_decile_bad(dec_df):
        if "decile" not in dec_df.columns:
            return np.nan
        bot_dec = int(dec_df["decile"].min())
        vals = dec_df.loc[dec_df["decile"] == bot_dec, "bad_rate"].values
        return float(vals[0]) if int(vals.shape[0]) > 0 else np.nan

    top_a = _top_decile_bad(res_a["overall"])
    top_b = _top_decile_bad(res_b["overall"])
    bot_a = _bottom_decile_bad(res_a["overall"])
    bot_b = _bottom_decile_bad(res_b["overall"])

    summary = pd.DataFrame([
        {"model": name_a, "auc": auc_a, "top_decile_bad_rate": top_a, "bottom_decile_bad_rate": bot_a, "seg_col_used": res_a.get("seg_col")},
        {"model": name_b, "auc": auc_b, "top_decile_bad_rate": top_b, "bottom_decile_bad_rate": bot_b, "seg_col_used": res_b.get("seg_col")}
    ])

    return summary, res_a, res_b

        print("No result dict found")
        return

    print("\nseg_col used")

    print("\noverall deciles (top)")

        print(overall_df.head(15))
    else:
        print(overall_df)

    print("\nby_segment keys")
    by_seg = res.get("by_segment")
    if isinstance(by_seg, dict):
        print(list(by_seg.keys()))
        if len(by_seg) > 0:
            first_key = list(by_seg.keys())[0]
            print("\nby_segment sample: " + str(first_key))
            print(by_seg[first_key].head(15))
    else:
        print(by_seg)
