def make_cluster_profile(
    X_out,
    labels,
    cols,
    inverse_log_cols=None,
    agg_func="mean"
):
    """
    Build a cluster profile table from X_out and labels.

    - X_out: dataframe of features used for clustering (after log1p / winsor).
    - labels: cluster labels (array-like, same length as X_out).
    - cols: list of columns to include in the profile.
    - inverse_log_cols: subset of cols that were log1p-transformed and
      should be inverse-transformed (exp - 1) for interpretability.
    - agg_func: "mean" or "median" (or any valid pandas agg name).

    Returns a dataframe:
      profile_log: aggregated values on the transformed scale
      profile_abs: same, but with inverse log1p applied to inverse_log_cols
    """
    if inverse_log_cols is None:
        inverse_log_cols = []

    # Copy to avoid mutating original
    df_clusters = X_out.copy()
    df_clusters["cluster"] = labels

    # Ensure we only use columns that exist
    cols_existing = [c for c in cols if c in df_clusters.columns]
    if not cols_existing:
        raise ValueError("None of the requested cols exist in X_out.")

    # Group by cluster and aggregate
    if agg_func == "mean":
        profile_log = df_clusters.groupby("cluster")[cols_existing].mean()
    else:
        profile_log = df_clusters.groupby("cluster")[cols_existing].agg(agg_func)

    # Make a copy for absolute-scale interpretation
    profile_abs = profile_log.copy()

    # Apply inverse log1p only to the specified columns that are present
    for col in inverse_log_cols:
        if col in profile_abs.columns:
            profile_abs[col] = np.expm1(profile_abs[col])

    return profile_log, profile_abs
