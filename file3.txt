def compute_iv(x,y,n_bins=10,bins=None,return_bins=False): """ Computes Information Value (IV) using quantile binning with guards and smoothing. Parameters ---------- feature : pd.Series Numeric predictor. target : pd.Series Binary target (0/1). n_bins : int Number of quantile bins. min_pos_rate : float Minimum fraction of positive class to compute IV. debug : bool If True, prints binning and IV info. Returns ------- float IV value (capped at 10.0) """ """ Compute IV for a single variable x vs binary target y. If bins is provided, use those; otherwise build bins with robust logic. """ # Align and drop NaNs data = pd.DataFrame({"x": x, "y": y}).dropna() x_non_null = data["x"] y_aligned = data["y"] # If constant feature, IV is 0 if x_non_null.nunique() <= 1: if return_bins: return 0.0, None return 0.0 # ------------------------ # Build or validate bins # ------------------------ if bins is None: # Build bins from scratch max_bins = min(n_bins, x_non_null.nunique()) try: # Try robust quantile-based bins binned, bin_edges = pd.qcut( x_non_null, q=max_bins, retbins=True, duplicates="drop", ) except ValueError: # Fallback: equal-width cuts across the range binned, bin_edges = pd.cut( x_non_null, bins=max_bins, retbins=True, include_lowest=True, ) else: # Use externally provided bin edges (e.g. from raw feature) bin_edges = np.unique(np.asarray(bins)) # Not enough unique edges → no real binning possible if bin_edges.shape[0] < 2: if return_bins: return 0.0, bin_edges return 0.0 # Ensure strictly increasing edges (guard against non-monotonic bins) diffs = np.diff(bin_edges) if np.any(diffs <= 0): eps = 1e-6 for i in range(1, len(bin_edges)): if bin_edges[i] <= bin_edges[i - 1]: bin_edges[i] = bin_edges[i - 1] + eps binned = pd.cut( x_non_null, bins=bin_edges, include_lowest=True, ) # ------------------------ # Compute WOE / IV # ------------------------ grouped = pd.DataFrame({"bin": binned, "y": y_aligned}).groupby("bin") bad = grouped["y"].sum() good = grouped["y"].count() - bad eps = 1e-6 bad_dist = (bad + eps) / (bad.sum() + eps) good_dist = (good + eps) / (good.sum() + eps) woe = np.log(bad_dist / good_dist) iv = ((bad_dist - good_dist) * woe).sum() if return_bins: return float(iv), bin_edges return float(iv)
