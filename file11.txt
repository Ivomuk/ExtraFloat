#  Cutoff sweep code

sweep_df = perf_eval[["never_loan_pd_like_num", "is_blacklisted"]].dropna().copy()
sweep_df = sweep_df.sort_values("never_loan_pd_like_num").reset_index(drop=True)

# Lower PD-like = safer, so approve if pd_like <= threshold
pd_vals = sweep_df["never_loan_pd_like_num"].values
y_vals = sweep_df["is_blacklisted"].values.astype(int)  n_total = int(sweep_df.shape[0])
n_black_total = int(y_vals.sum())
n_white_total = int(n_total - n_black_total)

baseline_black_rate = n_black_total / n_total

# Use percentile cutoffs as candidate operating points
pct_grid = list(range(5, 100, 5))  # 5%, 10%, ..., 95%
rows_out = []

for pct in pct_grid:
    idx = int(np.floor(n_total * pct / 100.0)) - 1
    if idx < 0:
        idx = 0
    if idx >= n_total:
        idx = n_total - 1

    thr = float(pd_vals[idx])

    approved_mask = (pd_vals <= thr)
    declined_mask = ~approved_mask

    n_app = int(approved_mask.sum())
    n_dec = int(declined_mask.sum())

    app_black = int(y_vals[approved_mask].sum())
    app_white = int(n_app - app_black)

    dec_black = int(y_vals[declined_mask].sum())
    dec_white = int(n_dec - dec_black)

    app_black_rate = app_black / n_app if n_app > 0 else np.nan
    app_white_rate = app_white / n_app if n_app > 0 else np.nan

    # Of all blacklists, what fraction did we catch by declining?
    blacklist_capture_decline = dec_black / n_black_total if n_black_total > 0 else np.nan

    # Of all whitelists, what fraction did we incorrectly decline?
    whitelist_decline_rate = dec_white / n_white_total if n_white_total > 0 else np.nan

    rows_out.append({
        "approval_pct": pct,
        "threshold_pd_like": thr,
        "approved_n": n_app,
        "approved_blacklist_rate": app_black_rate,
        "approved_whitelist_rate": app_white_rate,
        "declined_blacklist_capture": blacklist_capture_decline,
        "declined_whitelist_rate": whitelist_decline_rate,
        "improvement_vs_baseline_black_rate": baseline_black_rate - app_black_rate
    })

cutoff_sweep = pd.DataFrame(rows_out)

# Make it readable
cutoff_sweep_display = cutoff_sweep.copy()
for colnm in ["approved_blacklist_rate","approved_whitelist_rate","declined_blacklist_capture","declined_whitelist_rate","improvement_vs_baseline_black_rate"]:
    cutoff_sweep_display[colnm] = (cutoff_sweep_display[colnm] * 100.0).round(2)

cutoff_sweep_display["threshold_pd_like"] = cutoff_sweep_display["threshold_pd_like"].round(4)
