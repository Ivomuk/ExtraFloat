def compute_iv(x,y,n_bins=10,bins=None,return_bins=False):
    """
    Computes Information Value (IV) using quantile binning with guards and smoothing.

    Parameters
    ----------
    feature : pd.Series
        Numeric predictor.
    target : pd.Series
        Binary target (0/1).
    n_bins : int
        Number of quantile bins.
    min_pos_rate : float
        Minimum fraction of positive class to compute IV.
    debug : bool
        If True, prints binning and IV info.

    Returns
    -------
    float
        IV value (capped at 10.0)
    """
    

    """
    Compute IV for a single variable x vs binary target y.
    If bins is provided, use those; otherwise build bins with robust logic.
    """

    # Align and drop NaNs
    data = pd.DataFrame({"x": x, "y": y}).dropna()
    x_non_null = data["x"]
    y_aligned = data["y"]

    # If constant feature, IV is 0
    if x_non_null.nunique() <= 1:
        if return_bins:
            return 0.0, None
        return 0.0

    # ------------------------
    # Build or validate bins
    # ------------------------
    if bins is None:
        # Build bins from scratch
        max_bins = min(n_bins, x_non_null.nunique())

        try:
            # Try robust quantile-based bins
            binned, bin_edges = pd.qcut(
                x_non_null,
                q=max_bins,
                retbins=True,
                duplicates="drop",
            )
        except ValueError:
            # Fallback: equal-width cuts across the range
            binned, bin_edges = pd.cut(
                x_non_null,
                bins=max_bins,
                retbins=True,
                include_lowest=True,
            )
    else:
        # Use externally provided bin edges (e.g. from raw feature)
        bin_edges = np.unique(np.asarray(bins))

        # Not enough unique edges → no real binning possible
        if bin_edges.shape[0] < 2:
            if return_bins:
                return 0.0, bin_edges
            return 0.0

        # Ensure strictly increasing edges (guard against non-monotonic bins)
        diffs = np.diff(bin_edges)
        if np.any(diffs <= 0):
            eps = 1e-6
            for i in range(1, len(bin_edges)):
                if bin_edges[i] <= bin_edges[i - 1]:
                    bin_edges[i] = bin_edges[i - 1] + eps

        binned = pd.cut(
            x_non_null,
            bins=bin_edges,
            include_lowest=True,
        )

    # ------------------------
    # Compute WOE / IV
    # ------------------------
    grouped = pd.DataFrame({"bin": binned, "y": y_aligned}).groupby("bin")

    bad = grouped["y"].sum()
    good = grouped["y"].count() - bad

    eps = 1e-6
    bad_dist = (bad + eps) / (bad.sum() + eps)
    good_dist = (good + eps) / (good.sum() + eps)

    woe = np.log(bad_dist / good_dist)
    iv = ((bad_dist - good_dist) * woe).sum()

    if return_bins:
        return float(iv), bin_edges

    return float(iv)
	
	
	
	
	
def compute_iv_fixed_bins(x, y, n_bins=10, binning="quantile", return_bins=False, eps=1e-6):
    # Build bins from x (recommended: pass raw x here), then compute IV using those fixed bins
    data = pd.DataFrame({"x": x, "y": y}).dropna()
    x_nonnull = data["x"]
    y_aligned = data["y"]

    if x_nonnull.nunique(dropna=True) <= 1:
        if return_bins:
            return 0.0, None
        return 0.0

    max_bins = int(min(n_bins, x_nonnull.nunique(dropna=True)))

    if binning == "quantile":
        # Quantile bin edges from x; use qcut to get edges, then we will later use cut with edges
        try:
            _, bin_edges = pd.qcut(x_nonnull, q=max_bins, retbins=True, duplicates="drop")
        except ValueError:
            _, bin_edges = pd.cut(x_nonnull, bins=max_bins, retbins=True, include_lowest=True)
    else:
        # "uniform": equal width in x-space
        _, bin_edges = pd.cut(x_nonnull, bins=max_bins, retbins=True, include_lowest=True)

    bin_edges = np.unique(np.asarray(bin_edges))
    if bin_edges.shape[0] < 2:
        if return_bins:
            return 0.0, bin_edges
        return 0.0

    diffs = np.diff(bin_edges)
    if np.any(diffs <= 0):
        # enforce strictly increasing edges
        bin_edges = bin_edges.astype("float64")
        for i in range(1, len(bin_edges)):
            if bin_edges[i] <= bin_edges[i - 1]:
                bin_edges[i] = bin_edges[i - 1] + 1e-9

    binned = pd.cut(x_nonnull, bins=bin_edges, include_lowest=True)

    grouped = pd.DataFrame({"bin": binned, "y": y_aligned}).groupby("bin", observed=False)
    bad = grouped["y"].sum()
    good = grouped["y"].count() - bad

    bad_dist = (bad + eps) / (bad.sum() + eps)
    good_dist = (good + eps) / (good.sum() + eps)

    woe = np.log(bad_dist / good_dist)
    iv = ((bad_dist - good_dist) * woe).sum()

    if return_bins:
        return float(iv), bin_edges
    return float(iv)

def compute_iv_with_bins(x, y, bins, return_bins=False, eps=1e-6):

    data = pd.DataFrame({"x": x, "y": y}).dropna()
    x_nonnull = data["x"]
    y_aligned = data["y"]

    if x_nonnull.nunique(dropna=True) <= 1:
        if return_bins:

        return 0.0

    if bins is None:
        if return_bins:
            return 0.0, None
        return 0.0

    bin_edges = np.unique(np.asarray(bins))
    if bin_edges.shape[0] < 2:
        if return_bins:

    diffs = np.diff(bin_edges)
    if np.any(diffs <= 0):
        bin_edges = bin_edges.astype("float64")
        for i in range(1, len(bin_edges)):
            if bin_edges[i] <= bin_edges[i - 1]:
                bin_edges[i] = bin_edges[i - 1] + 1e-9

    binned = pd.cut(x_nonnull, bins=bin_edges, include_lowest=True)

    grouped = pd.DataFrame({"bin": binned, "y": y_aligned}).groupby("bin", observed=False)
    bad = grouped["y"].sum()
    good = grouped["y"].count() - bad

    bad_dist = (bad + eps) / (bad.sum() + eps)
    good_dist = (good + eps) / (good.sum() + eps)

    woe = np.log(bad_dist / good_dist)

    if return_bins:

    return float(iv)


def iv_filter_phase_2(
    X_train_raw,
    X_train_transformed,
    y_train,
    min_iv=0.02,
    n_bins=10,
    pd_feature_blacklist=None,
    forbidden_feature_patterns=None,
    target_col="bad_state",
    debug=False,
):
    """
    Performs IV filtering on TRAIN ONLY using the enhanced compute_iv function.

    Parameters
    ----------
    X_train_raw : pd.DataFrame
        Raw numeric features before transformation.
    X_train_transformed : pd.DataFrame
        Transformed numeric features (e.g., log, capped).
    y_train : pd.Series
        Binary target (0/1).
    min_iv : float
        Minimum IV threshold to select features.
    n_bins : int
        Number of quantile bins for IV calculation.
    debug : bool
        If True, prints IV computation details per feature.

    Returns
    -------
    selected_features : list
        Features with IV >= min_iv.
    iv_df : pd.DataFrame
        DataFrame with iv_before, iv_after, and uplift for all features.
    """

    
  

    if pd_feature_blacklist is None:
        pd_feature_blacklist = []
    if forbidden_feature_patterns is None:
        forbidden_feature_patterns = []

    blacklist_low = set([str(c).lower() for c in pd_feature_blacklist])
    patterns_low = [str(p).lower() for p in forbidden_feature_patterns]
    target_low = str(target_col).lower()

    transformed_cols = list(X_train_transformed.columns)
    raw_cols_set = set(X_train_raw.columns) if X_train_raw is not None else set()

    iv_records = []

    for col in transformed_cols:
        col_low = str(col).lower()

        if (
            col_low == target_low
            or col_low in blacklist_low
            or any(p in col_low for p in patterns_low)
        ):
            if debug:
                print("[SKIP] Forbidden column: " + str(col))
            continue

        x_after = X_train_transformed[col]
        iv_after = compute_iv(x_after, y_train, n_bins=n_bins)

        if (X_train_raw is not None) and (col in raw_cols_set):
            x_before = X_train_raw[col]
            iv_before = compute_iv(x_before, y_train, n_bins=n_bins)
            iv_uplift = iv_after - iv_before
        else:
            x_before = None
            iv_before = np.nan
            iv_uplift = np.nan

        if debug and col == "average_balance" and x_before is not None:
            eq_rate = (x_before.fillna(0).values == x_after.fillna(0).values).mean()

            # bin invariance check if compute_iv is quantile/rank based
            try:
                b_before = pd.qcut(x_before, q=n_bins, labels=False, duplicates="drop")
                b_after = pd.qcut(x_after, q=n_bins, labels=False, duplicates="drop")
                same_bin_rate = (b_before.fillna(-1).astype(int).values == b_after.fillna(-1).astype(int).values).mean()
            except Exception:
                same_bin_rate = np.nan

            print("DBG average_balance eq_rate:", eq_rate)
            print("DBG average_balance same_bin_rate:", same_bin_rate)
            print("DBG average_balance iv_before:", iv_before)
            print("DBG average_balance iv_after:", iv_after)
            print("DBG average_balance uplift:", iv_uplift)

        iv_records.append(
            {
                "feature": col,
                "iv_before": iv_before,
                "iv_after": iv_after,
                "iv_uplift": iv_uplift,
            }
        )

    iv_table = (
        pd.DataFrame(iv_records)
        .sort_values("iv_after", ascending=False)
        .reset_index(drop=True)
    )

    min_uplift = -0.05
    selected_mask = (
        (iv_table["iv_after"] >= min_iv)
        & (iv_table["iv_uplift"].fillna(min_uplift) >= min_uplift)
    )
    selected_features = iv_table.loc[selected_mask, "feature"].tolist()

    if debug:
        print(
            "\nSelected " + str(len(selected_features))
            + " features with IV >= " + str(min_iv)
            + " and uplift >= " + str(min_uplift)
        )
        print("\nTop 20 SELECTED features by IV after transformation:")
        print(iv_table.loc[selected_mask].head(20)[["feature", "iv_after", "iv_before", "iv_uplift"]])

    return selected_features, iv_table

def _iv_from_bins(bin_ids, y_arr, eps=1e-6):
    # Compute IV given precomputed bin assignments (labels), with NaNs allowed
    bin_series = pd.Series(bin_ids)
    y_series = pd.Series(y_arr)

    mask = bin_series.notna() & y_series.notna()
    if mask.sum() == 0:
        return 0.0

    df_tmp = pd.DataFrame({"bin": bin_series[mask], "y": y_series[mask]})
    grouped = df_tmp.groupby("bin", observed=False)

    bad = grouped["y"].sum()
    good = grouped["y"].count() - bad

    bad_dist = (bad + eps) / (bad.sum() + eps)
    good_dist = (good + eps) / (good.sum() + eps)

    woe = np.log(bad_dist / good_dist)
    iv_val = ((bad_dist - good_dist) * woe).sum()
    return float(iv_val)


def _make_rank_bins(x, n_bins=10, binning="quantile"):
    # Produce stable bin assignments based on x's distribution (in raw space)
    x_series = pd.Series(x)
    x_nonnull = x_series.dropna()
    if x_nonnull.nunique(dropna=True) <= 1:
        return pd.Series([np.nan] * len(x_series), index=x_series.index)

    max_bins = int(min(n_bins, x_nonnull.nunique(dropna=True)))

    if binning == "quantile":
        try:
            bin_ids = pd.qcut(x_series, q=max_bins, labels=False, duplicates="drop")
        except ValueError:
            bin_ids = pd.cut(x_series, bins=max_bins, labels=False, include_lowest=True)
    else:
        bin_ids = pd.cut(x_series, bins=max_bins, labels=False, include_lowest=True)

    return bin_ids


def iv_filter_phase_2_separability_sensitive(
    X_train_raw,
    X_train_transformed,
    y_train,
    min_iv=0.02,
    n_bins=10,
    pd_feature_blacklist=None,
    forbidden_feature_patterns=None,
    target_col="bad_state",
    min_uplift=-0.05,
    binning="quantile",
    debug=False,
):
    if pd_feature_blacklist is None:
        pd_feature_blacklist = []
    if forbidden_feature_patterns is None:
        forbidden_feature_patterns = []

    blacklist_low = set([str(c).lower() for c in pd_feature_blacklist])
    patterns_low = [str(p).lower() for p in forbidden_feature_patterns]
    target_low = str(target_col).lower()

    transformed_cols = list(X_train_transformed.columns)
    raw_cols_set = set(X_train_raw.columns) if X_train_raw is not None else set()

    # normalize y once
    y_vec = y_train
    if isinstance(y_vec, pd.DataFrame) and y_vec.shape[1] == 1:
        y_vec = y_vec.iloc[:, 0]
    if isinstance(y_vec, (pd.Series, pd.Index)):
        y_arr = y_vec.to_numpy()
    else:
        y_arr = np.asarray(y_vec)

    def make_rank_bins(x_series):
        x_s = pd.Series(x_series)
        x_nonnull = x_s.dropna()
        if x_nonnull.nunique(dropna=True) <= 1:
            return pd.Series([np.nan] * len(x_s), index=x_s.index)
        max_bins = int(min(n_bins, x_nonnull.nunique(dropna=True)))
        if binning == "quantile":
            try:
                return pd.qcut(x_s, q=max_bins, labels=False, duplicates="drop")
            except Exception:
                return pd.cut(x_s, bins=max_bins, labels=False, include_lowest=True)
        return pd.cut(x_s, bins=max_bins, labels=False, include_lowest=True)

    def iv_from_bin_ids(bin_ids, y_vals, eps=1e-6):
        bin_s = pd.Series(bin_ids)
        y_s = pd.Series(y_vals)
        ok = bin_s.notna() & y_s.notna()
        if ok.sum() == 0:
            return 0.0
        tmp = pd.DataFrame({"bin": bin_s[ok], "y": y_s[ok]})
        grp = tmp.groupby("bin", observed=False)["y"]
        bad = grp.sum()
        cnt = grp.count()
        good = cnt - bad
        bad_dist = (bad + eps) / (bad.sum() + eps)
        good_dist = (good + eps) / (good.sum() + eps)
        woe = np.log(bad_dist / good_dist)
        return float(((bad_dist - good_dist) * woe).sum())

    def fixed_edges_uplift_from_raw(x_before_series, x_after_series):
        # Diagnostic:
        # - Learn bin edges in RAW value space
        # - Compute IV in raw using those edges
        # - For transformed values, map them back onto RAW scale via percentile matching
        #   (prevents unit-mismatch artifacts when transform is log/scale/etc.)

        x_before_s = pd.Series(x_before_series)
        x_after_s = pd.Series(x_after_series)

        x_before_nonnull = x_before_s.dropna()
        if x_before_nonnull.nunique(dropna=True) <= 1:
            return np.nan, np.nan, np.nan          max_bins_edges = int(min(n_bins, x_before_nonnull.nunique(dropna=True)))

        if binning == "quantile":
            try:
                _, raw_edges = pd.qcut(
                    x_before_nonnull,
                    q=max_bins_edges,
                    retbins=True,
                    duplicates="drop",
                )
            except Exception:
                raw_edges = np.linspace(
                    float(x_before_nonnull.min()),
                    float(x_before_nonnull.max()),
                    num=max_bins_edges + 1,
                )
        else:
            raw_edges = np.linspace(
                float(x_before_nonnull.min()),
                float(x_before_nonnull.max()),
                num=max_bins_edges + 1,
            )

        raw_edges = np.unique(raw_edges)
        if raw_edges.shape[0] < 2:
            return np.nan, np.nan, np.nan

        # before
        bin_ids_before = pd.cut(x_before_s, bins=raw_edges, labels=False, include_lowest=True)
        iv_b = iv_from_bin_ids(bin_ids_before, y_arr)

        # after -> percentile match -> raw scale
        ok_after = x_after_s.notna()
        if ok_after.sum() == 0:
            return float(iv_b), np.nan, np.nan

        after_pct = x_after_s[ok_after].rank(method="average", pct=True).to_numpy()

        raw_sorted = np.sort(x_before_nonnull.to_numpy())
        raw_n = raw_sorted.shape[0]
        if raw_n <= 1:
            return float(iv_b), np.nan, np.nan

        raw_idx = np.clip((after_pct * (raw_n - 1)).round().astype(int), 0, raw_n - 1)
        after_mapped = pd.Series(np.nan, index=x_after_s.index)
        after_mapped.loc[ok_after] = raw_sorted[raw_idx]

        bin_ids_after = pd.cut(after_mapped, bins=raw_edges, labels=False, include_lowest=True)
        iv_a = iv_from_bin_ids(bin_ids_after, y_arr)

        return float(iv_b), float(iv_a), float(iv_a - iv_b)

    iv_records = []
    for col in transformed_cols:
        col_low = str(col).lower()
        if (col_low == target_low) or (col_low in blacklist_low) or any(p in col_low for p in patterns_low):
            if debug:
                print("[SKIP] Forbidden column: " + str(col))
            continue

        x_after = X_train_transformed[col]

        # Selection IV: bins in transformed space (UNCHANGED)
        iv_after = compute_iv(x_after, y_train, n_bins=n_bins)

        if (X_train_raw is not None) and (col in raw_cols_set):
            x_before = X_train_raw[col]

            iv_before = compute_iv(x_before, y_train, n_bins=n_bins)
            iv_uplift = iv_after - iv_before

            # Fixed-rank diagnostic IV (UNCHANGED)
            bin_ids = make_rank_bins(x_before)
            iv_before_fixed_rank = iv_from_bin_ids(bin_ids, y_arr)
            bin_ids_after = make_rank_bins(x_after)
            iv_after_fixed_rank = iv_from_bin_ids(bin_ids_after, y_arr)
            iv_uplift_fixed_rank = iv_after_fixed_rank - iv_before_fixed_rank

            # Fixed-edges diagnostic IV (IMPROVED)
            iv_before_fixed_edges, iv_after_fixed_edges, iv_uplift_fixed_edges = fixed_edges_uplift_from_raw(x_before, x_after
            )
        else:
            iv_before = np.nan
            iv_uplift = np.nan
            iv_before_fixed_rank = np.nan
            iv_after_fixed_rank = np.nan
            iv_uplift_fixed_rank = np.nan
            iv_before_fixed_edges = np.nan
            iv_after_fixed_edges = np.nan
            iv_uplift_fixed_edges = np.nan

        iv_records.append(
            {
                "feature": col,
                "iv_before": float(iv_before) if pd.notna(iv_before) else np.nan,
                "iv_after": float(iv_after) if pd.notna(iv_after) else np.nan,
                "iv_uplift": float(iv_uplift) if pd.notna(iv_uplift) else np.nan,
                "iv_before_fixed_rank": float(iv_before_fixed_rank) if pd.notna(iv_before_fixed_rank) else np.nan,
                "iv_after_fixed_rank": float(iv_after_fixed_rank) if pd.notna(iv_after_fixed_rank) else np.nan,
                "iv_uplift_fixed_rank": float(iv_uplift_fixed_rank) if pd.notna(iv_uplift_fixed_rank) else np.nan,
                "iv_before_fixed_edges": float(iv_before_fixed_edges) if pd.notna(iv_before_fixed_edges) else np.nan,
                "iv_after_fixed_edges": float(iv_after_fixed_edges) if pd.notna(iv_after_fixed_edges) else np.nan,
                "iv_uplift_fixed_edges": float(iv_uplift_fixed_edges) if pd.notna(iv_uplift_fixed_edges) else np.nan,
            }
        )

    iv_table = pd.DataFrame(iv_records)
    if iv_table.shape[0] == 0:
        return [], iv_table

    iv_table = iv_table.sort_values("iv_after", ascending=False).reset_index(drop=True)

    selected_mask = (iv_table["iv_after"] >= min_iv) & (iv_table["iv_uplift"].fillna(min_uplift) >= min_uplift)
    selected_features = iv_table.loc[selected_mask, "feature"].tolist()

    if debug:
        print(
            "\nSelected " + str(len(selected_features)) +
            " features with IV >= " + str(min_iv) +
            " and uplift >= " + str(min_uplift)
        )
        print("\nTop 20 SELECTED features by IV after transformation:")
        print(iv_table.loc[selected_mask].head(20)[["feature", "iv_after", "iv_before", "iv_uplift"]])

        harmed = (
            iv_table.loc[iv_table["iv_uplift"].notna()]
            .sort_values("iv_uplift", ascending=True)
            .head(10)
        )
        if harmed.shape[0] > 0:
            print("\nBottom 10 by uplift (selection IV):")
            print(harmed[["feature", "iv_after", "iv_before", "iv_uplift"]])

        harmed_fixed = (
            iv_table.loc[iv_table["iv_uplift_fixed_rank"].notna()]
            .sort_values("iv_uplift_fixed_rank", ascending=True)
            .head(10)
        )
        if harmed_fixed.shape[0] > 0:
            print("\nBottom 10 by uplift (fixed-rank diagnostic):")
            print(harmed_fixed[["feature", "iv_after_fixed_rank", "iv_before_fixed_rank", "iv_uplift_fixed_rank"]])

def prune_post_transform_features(df, pd_features, min_unique=2, drop_all_nan=True, debug=True):
    # Only consider features that exist in df
    present_feats = [c for c in pd_features if c in df.columns]

    # Track columns to drop
    drop_cols = []

    for c in present_feats:
        s = df[c]

        # Optionally drop features that are entirely missing
        if drop_all_nan and s.notna().sum() == 0:
            drop_cols.append(c)
            continue

        # Drop features with < min_unique non-null unique values
        nunq = s.nunique(dropna=True)
        if nunq < min_unique:
            drop_cols.append(c)

    # Drop and return updated feature list
    if drop_cols:
        df = df.drop(columns=drop_cols)
    kept_features = [c for c in pd_features if c in df.columns]

    if debug:
        print("Post-transform pruned:", len(drop_cols))
        print(sorted(drop_cols)[:50])
        print("Kept features:", len(kept_features))

    return df, kept_features, drop_cols
