def leakage_audit_phase_2_2( df_pd, snapshot_col="snapshot_dt", txn_max_col="max_txn_dt", label_cols=("bad_state", "current_dpd", "ever_written_off", "ever_in_collections"), hard_fail=True, corr_high=0.85, corr_med=0.65, dominance_high=0.70, max_corr_cols=500, verbose=True ): audit_rows = [] # ------------------------- # 0) Basic guards # ------------------------- if not isinstance(df_pd, pd.DataFrame): raise TypeError("df_pd must be a pandas DataFrame") # ------------------------- # 1) Target leakage check # ------------------------- for col_name in df_pd.columns: if col_name in label_cols and col_name != "bad_state": audit_rows.append({ "check": "TARGET_LEAKAGE", "column": col_name, "severity": "HIGH", "message": "Label-related column present in feature table: " + str(col_name) }) # ------------------------- # 2) Future-time leakage # ------------------------- if snapshot_col in df_pd.columns and txn_max_col in df_pd.columns: snap_ser = df_pd[snapshot_col] txn_ser = df_pd[txn_max_col] if not pd.api.types.is_datetime64_any_dtype(snap_ser): snap_ser = pd.to_datetime(snap_ser, errors="coerce") if not pd.api.types.is_datetime64_any_dtype(txn_ser): txn_ser = pd.to_datetime(txn_ser, errors="coerce") future_mask = (txn_ser.notna()) & (snap_ser.notna()) & (txn_ser > snap_ser) if bool(future_mask.any()): audit_rows.append({ "check": "TIME_LEAKAGE", "column": txn_max_col, "severity": "HIGH", "message": str(int(future_mask.sum())) + " rows have transactions after snapshot date" }) # ------------------------- # 3) Label-proxy dominance via correlation # ------------------------- if "bad_state" in df_pd.columns: y = pd.to_numeric(df_pd["bad_state"], errors="coerce") if y.notna().sum() > 0: numeric_cols = [ c for c in df_pd.columns if c != "bad_state" and pd.api.types.is_numeric_dtype(df_pd[c]) ] # Keep runtime sane on very wide tables if len(numeric_cols) > max_corr_cols: numeric_cols = numeric_cols[:max_corr_cols] for col_name in numeric_cols: x = pd.to_numeric(df_pd[col_name], errors="coerce") valid_mask = x.notna() & y.notna() if int(valid_mask.sum()) < 50: continue corr_val = x.loc[valid_mask].corr(y.loc[valid_mask]) if pd.isna(corr_val): continue abs_corr = float(abs(corr_val)) if abs_corr > corr_high: audit_rows.append({ "check": "LABEL_PROXY", "column": col_name, "severity": "HIGH", "message": "Very high correlation with bad_state: " + str(round(corr_val, 3)) }) elif abs_corr > corr_med: audit_rows.append({ "check": "LABEL_PROXY", "column": col_name, "severity": "MEDIUM", "message": "High correlation with bad_state: " + str(round(corr_val, 3)) }) # ------------------------- # 4) Dominant binary separators # ------------------------- if "bad_state" in df_pd.columns: y = pd.to_numeric(df_pd["bad_state"], errors="coerce") if y.notna().sum() > 0: for col_name in df_pd.columns: if col_name == "bad_state": continue ser = df_pd[col_name] # Only check true binary-like columns nunq = ser.dropna().nunique() if int(nunq) != 2: continue # Compare rates for the two states as they appear vals = list(pd.Series(ser.dropna().unique()).sort_values()) if len(vals) != 2: continue v0 = vals[0] v1 = vals[1] mask0 = ser == v0 mask1 = ser == v1 if int(mask0.sum()) < 30 or int(mask1.sum()) < 30: continue rate_0 = float(y.loc[mask0].mean()) rate_1 = float(y.loc[mask1].mean()) jump = float(abs(rate_1 - rate_0)) if np.isfinite(jump) and jump > dominance_high: audit_rows.append({ "check": "BINARY_DOMINANCE", "column": col_name, "severity": "HIGH", "message": "Bad rate jump " + str(round(jump, 3)) + " between " + str(v0) + " and " + str(v1) }) # ------------------------- # Final report # ------------------------- audit_report = pd.DataFrame(audit_rows) if verbose: if not audit_report.empty: sev_order = {"HIGH": 0, "MEDIUM": 1, "LOW": 2} audit_report_tmp = audit_report.copy() audit_report_tmp["_sev_order"] = audit_report_tmp["severity"].map(sev_order).fillna(9).astype(int) audit_report_tmp = audit_report_tmp.sort_values(["_sev_order", "check", "column"]).drop(columns=["_sev_order"]) print("\nLEAKAGE AUDIT WARNINGS") print(audit_report_tmp) else: print("Leakage audit passed — no red flags detected") if (not audit_report.empty) and bool(hard_fail): high_df = audit_report[audit_report["severity"] == "HIGH"].copy() if not high_df.empty: # Format a compact, readable message high_df = high_df.sort_values(["check", "column"]) msg_lines = [] msg_lines.append("HARD LEAKAGE DETECTED — FIX BEFORE MODELING") msg_lines.append("HIGH findings (" + str(len(high_df)) + "):") for _, row in high_df.iterrows(): msg_lines.append( "- " + str(row["check"]) + " | " + str(row["column"]) + " | " + str(row["message"]) ) raise ValueError("\n".join(msg_lines)) return audit_report


def compute_bad_flags(df_pd): """ Computes: - hard_bad_flag → short-term deterioration / monitoring - bad_state → PD target (forward-looking 30D) Requires: bad_state_30D from SQL. """ df_pd = df_pd.copy() # HARD BAD FLAG (monitoring) repayment_coverage_1M = df_pd.get("repayment_coverage_1M", pd.Series(1, index=df_pd.index)) penalties_1M = df_pd.get("penalties_1M", pd.Series(0, index=df_pd.index)) repayment_val_1M = df_pd.get("repayment_val_1M", pd.Series(1, index=df_pd.index)) df_pd["hard_bad_flag"] = ( (repayment_coverage_1M < 0.3) & (penalties_1M > 0) & (repayment_val_1M == 0) ).astype(int) # PD TARGET (forward-looking) if "bad_state_30D" not in df_pd.columns: raise RuntimeError("Missing required column bad_state_30D (should be generated by SQL).") df_pd["bad_state"] = (df_pd["bad_state_30D"].fillna(0) > 0).astype(int) # Guardrails assert set(df_pd["bad_state"].unique()).issubset({0, 1}), "bad_state contains values outside {0,1}" assert not df_pd["bad_state"].isna().any(), "bad_state contains NaNs" return df_pd
