from sklearn.linear_model import LinearRegression
from sklearn.ensemble import RandomForestRegressor
from sklearn.model_selection import train_test_split

def test_nonlinearity_sampled(df, target_col=None, sample_size=20000):
        """
        Lightweight + RAM-safe non-linearity test using sampling.
    
        Tests included:
        1. PCA (5 components) → checks linear structure strength
        2. KMeans vs GMM (diag covariance) → clustering linearity
        3. Linear Regression vs Random Forest → predictive non-linearity
    
        df : pandas dataframe
        target_col : name of the target column or None (RF/LR test skipped)
        sample_size : max rows to sample
        """
    
        # ------------------------------------------------------------------
        # STEP 1 — SAMPLE
        # ------------------------------------------------------------------
        n = len(df)
        if n > sample_size:
            idx = np.random.choice(n, size=sample_size, replace=False)
            df_sample = df.iloc[idx]
        else:
            df_sample = df.copy()
    
        # Separate X and y
        if target_col is not None:
            y = df_sample[target_col].values
            X = df_sample.drop(columns=[target_col]).values
        else:
            X = df_sample.values
            y = None
    
        results = {}
    
        # ------------------------------------------------------------------
        # STEP 2 — PCA (linear structure)
        # ------------------------------------------------------------------
        pca = PCA(n_components=min(5, X.shape[1]))
        pca.fit(X)
    
        var3 = np.sum(pca.explained_variance_ratio_[:3])
        results["pca_variance_3_components"] = var3
    
        # ------------------------------------------------------------------
        # STEP 3 — KMeans vs GMM (non-linear clustering)
        # ------------------------------------------------------------------
        k = 4
    
        kmeans = KMeans(n_clusters=k, random_state=42, n_init="auto").fit(X)
    
        gmm = GaussianMixture(
            n_components=k,
            random_state=42,
            covariance_type="diag",  # RAM-SAFE
            max_iter=100
        ).fit(X)
    
        labels_gmm = gmm.predict(X)
    
        sil_km = silhouette_score(X, kmeans.labels_)
        sil_gmm = silhouette_score(X, labels_gmm)
    
        results["silhouette_kmeans"] = sil_km
        results["silhouette_gmm"] = sil_gmm
        results["gmm_minus_kmeans"] = sil_gmm - sil_km
    
        # ------------------------------------------------------------------
        # STEP 4 — Regression Test (if target_col provided)
        # ------------------------------------------------------------------
        if target_col is not None:
            X_train, X_test, y_train, y_test = train_test_split(
                X, y, test_size=0.25, random_state=42
            )
    
            lr = LinearRegression().fit(X_train, y_train)
            rf = RandomForestRegressor(
                n_estimators=50,   # small
                max_depth=6,      # small
                n_jobs=-1,
                random_state=42
            ).fit(X_train, y_train)
    
            results["r2_linear"] = lr.score(X_test, y_test)
            results["r2_random_forest"] = rf.score(X_test, y_test)
            results["rf_minus_lr"] = results["r2_random_forest"] - results["r2_linear"]
    
        return results
