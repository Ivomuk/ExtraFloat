            .astype(str).str.strip().str.replace(r"\D", "", regex=True).astype("int64"))
    
    # Merge whitelist (CommissionDecision + agent_category)
    .merge(
        df_xtrafloat_whitelist
        .assign(agent_msisdn=lambda df: df["agent_msisdn"]
                .astype(str).str.strip().str.replace(r"\D", "", regex=True).astype("int64"))
        [["agent_msisdn", "CommissionDecision", "agent_category"]],
        on="agent_msisdn",
        how="left"
    )
    .rename(columns={
        "CommissionDecision": "CommissionDecision_wl",
        "agent_category": "agent_category_wl"
    })
    
    # Merge blacklist (CommissionDecision + agent_category)
    .merge(
        df_xtrafloat_blacklist
        .assign(agent_msisdn=lambda df: df["agent_msisdn"]
                .astype(str).str.strip().str.replace(r"\D", "", regex=True).astype("int64"))
        [["agent_msisdn", "CommissionDecision", "agent_category"]],
        on="agent_msisdn",
        how="left"
    )
    .rename(columns={
        "CommissionDecision": "CommissionDecision_bl",
        "agent_category": "agent_category_bl"
    })
    
    # Final single columns: whitelist takes priority
    .assign(
        CommissionDecision=lambda df: df["CommissionDecision_wl"].combine_first(df["CommissionDecision_bl"]),
        agent_category=lambda df: df["agent_category_wl"].combine_first(df["agent_category_bl"])
    )
    
    # Drop temporary columns
    .drop(columns=["CommissionDecision_wl", "CommissionDecision_bl", "agent_category_wl", "agent_category_bl"])

def cluster_purity(df, cluster_col, label_col):
    tab = pd.crosstab(df[cluster_col], df[label_col])
    if tab.shape[0] == 0:
        # Handle empty df (e.g. no blacklist rows)
        return pd.Series(dtype="float64"), float("")
    max_in_row = tab.max(axis=1)
    purity_per_cluster = (max_in_row / tab.sum(axis=1)).rename("purity")
    overall_purity = max_in_row.sum() / tab.values.sum()
    return purity_per_cluster, overall_purity

# Helper to compute ARI safely by dropping NaNs
def safe_ari(labels_true, labels_pred):
    mask = (~labels_true.isna()) & (~labels_pred.isna())
    if mask.sum() == 0:
        return np.nan
    return adjusted_rand_score(labels_true[mask], labels_pred[mask])

# Helper: purity for a given dataframe slice
def cluster_purity_table(df, cluster_col="ensemble_cluster", label_col="agent_category", suffix=""):
    # crosstab: clusters x labels
    tab = pd.crosstab(df[cluster_col], df[label_col])
    if tab.shape[0] == 0:
        # no rows in this slice
        return pd.DataFrame(columns=[
            "dominant_agent_category" + suffix,
            "purity" + suffix
        ])

    # majority label per row
    dominant = tab.idxmax(axis=1)
    max_in_row = tab.max(axis=1)
    purity = max_in_row / tab.sum(axis=1)

    out = pd.DataFrame({
        "ensemble_cluster": tab.index,
        "dominant_agent_category" + suffix: dominant.values,
        "purity" + suffix: purity.values
    }).set_index("ensemble_cluster")

    return out

def bt_counts_for_slice(df, cluster_col="ensemble_cluster", label_col="agent_category", suffix="_all"):
    # Group by cluster and count rows
    group = df.groupby(cluster_col)

    counts = group[label_col].agg(
        total_agents="count",
        bt_count=lambda x: (x == "Below Threshold").sum()
    )

    counts["non_bt_count"] = counts["total_agents"] - counts["bt_count"]
    counts["bt_share"] = counts["bt_count"] / counts["total_agents"]

    # Rename with suffix so we can join cleanly later
    counts = counts.rename(columns={
        "total_agents": "total_agents" + suffix,
        "bt_count": "bt_count" + suffix,
        "non_bt_count": "non_bt_count" + suffix,
        "bt_share": "bt_share" + suffix
    })

    return counts
