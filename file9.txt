# PD Calibration Phase 1 + Phase 2 pipeline as reusable functions.
# It will fail-closed on small samples / too few bads, enforce monotonic calibration via isotonic regression,
# and require calibrated PD coverage before producing policy CSV outputs.

def _standardize_scored_df(scored_df, model_key):
    df_local = scored_df.copy()

    # Label
    if 'y' not in df_local.columns and 'bad_state' in df_local.columns:
        df_local['y'] = pd.to_numeric(df_local['bad_state'], errors='coerce')
    df_local['y'] = (pd.to_numeric(df_local['y'], errors='coerce') > 0.5).astype(int)

    # ID
    if 'agent_msisdn' not in df_local.columns:
        df_local['agent_msisdn'] = df_local.index.astype(str)
    else:
        df_local['agent_msisdn'] = df_local['agent_msisdn'].astype(str)

    # Score
    score_col_local = None
    if model_key == 'xgb':
        if 'xgb_score' in df_local.columns:
            score_col_local = 'xgb_score'
        elif 'score' in df_local.columns:
            score_col_local = 'score'
    if model_key == 'lgb':
        if 'lgb_score' in df_local.columns:
            score_col_local = 'lgb_score'
        elif 'score' in df_local.columns:
            score_col_local = 'score'

    if score_col_local is None:
        # fallback: first column containing 'score'
        score_cands = [c for c in df_local.columns if 'score' in str(c).lower()]
        if len(score_cands) > 0:
            score_col_local = score_cands[0]

    if score_col_local is None:
        raise ValueError('No score column found for model ' + str(model_key))

    df_local['model_score'] = pd.to_numeric(df_local[score_col_local], errors='coerce')

    df_local = df_local.replace([np.inf, -np.inf], np.nan)
    df_local = df_local[['agent_msisdn','y','model_score']].dropna()

    return df_local, score_col_local


def build_pd_calibration_map(scored_df, model_key, n_bins=50, min_n=5000, min_bads=50):
    df_local, score_col_local = _standardize_scored_df(scored_df, model_key)

    n_obs = int(df_local.shape[0])
    n_bads = int(df_local['y'].sum())
    event_rate = float(df_local['y'].mean())

    if n_obs < int(min_n):
        raise ValueError('Fail-closed: n=' + str(n_obs) + ' < min_n=' + str(min_n) + ' for model ' + model_key)
    if n_bads < int(min_bads):
        raise ValueError('Fail-closed: bads=' + str(n_bads) + ' < min_bads=' + str(min_bads) + ' for model ' + model_key)
    if event_rate <= 0.0 or event_rate >= 1.0:
        raise ValueError('Fail-closed: event_rate=' + str(event_rate) + ' is degenerate for model ' + model_key)

    # Determine risk orientation
    corr_val = float(pd.Series(df_local['model_score']).corr(pd.Series(df_local['y'])))
    if np.isnan(corr_val):
        raise ValueError('Fail-closed: correlation undefined (label constant?) for model ' + model_key)
    ascending_risk = bool(corr_val > 0)

    # Bin by score quantiles
    df_local = df_local.sort_values('model_score', ascending=True).reset_index(drop=True)
    try:
        df_local['bin'] = pd.qcut(df_local['model_score'], q=int(n_bins), labels=False, duplicates='drop')
    except Exception:
        # fallback: fewer bins
        df_local['bin'] = pd.qcut(df_local['model_score'], q=20, labels=False, duplicates='drop')

    bin_tbl = df_local.groupby('bin', as_index=False).agg(
        n=('y','size'),
        bad_rate=('y','mean'),
        score_min=('model_score','min'),
        score_max=('model_score','max'),
        score_mean=('model_score','mean')
    ).sort_values('score_mean').reset_index(drop=True)

    # Apply isotonic regression for monotonic PD
    x_vals = bin_tbl['score_mean'].to_numpy()
    y_vals = bin_tbl['bad_rate'].to_numpy()

    # If higher score means higher risk, PD should be increasing in score_mean, else decreasing.
    if ascending_risk:
        iso = IsotonicRegression(increasing=True, out_of_bounds='clip')
        pd_hat = iso.fit_transform(x_vals, y_vals)
    else:
        iso = IsotonicRegression(increasing=False, out_of_bounds='clip')
        pd_hat = iso.fit_transform(x_vals, y_vals)

    # Clip PD away from 0/1 for operational stability
    eps = 1e-6
    pd_hat = np.clip(pd_hat, eps, 1.0 - eps)  
    bin_tbl['pd'] = pd_hat
    bin_tbl['model'] = model_key
    bin_tbl['ascending_risk'] = ascending_risk
    bin_tbl['score_col'] = score_col_local

    # Construct mapping edges as half-open intervals [min, max) except last bin inclusive.
    map_tbl = bin_tbl[['model','score_min','score_max','pd','ascending_risk','score_col','n','bad_rate']].copy()

    # Make score_max strictly increasing to avoid gaps due to floating point ties
    map_tbl = map_tbl.sort_values(['score_min','score_max']).reset_index(drop=True)

    return map_tbl


def attach_cal_pd(scored_df, cal_map_tbl, model_key, min_coverage=0.98):
    df_local, score_col_local = _standardize_scored_df(scored_df, model_key)

    cal_sub = cal_map_tbl[cal_map_tbl['model'].astype(str) == str(model_key)].copy()
    if cal_sub.shape[0] == 0:
        raise ValueError('Fail-closed: no calibration mapping for model ' + model_key)

    cal_sub['score_min'] = pd.to_numeric(cal_sub['score_min'], errors='coerce')
    cal_sub['score_max'] = pd.to_numeric(cal_sub['score_max'], errors='coerce')
    cal_sub['pd'] = pd.to_numeric(cal_sub['pd'], errors='coerce')
    cal_sub = cal_sub.dropna(subset=['score_min','score_max','pd']).sort_values(['score_min','score_max']).reset_index(drop=True)

    score_vals = pd.to_numeric(df_local['model_score'], errors='coerce').to_numpy()
    pd_vals = np.full(int(df_local.shape[0]), np.nan)

    min_edge = float(cal_sub['score_min'].min())
    max_edge = float(cal_sub['score_max'].max())

    for i in range(int(cal_sub.shape[0])):
        lo = float(cal_sub.loc[i, 'score_min'])
        hi = float(cal_sub.loc[i, 'score_max'])
        if i < int(cal_sub.shape[0]) - 1:
            msk = (score_vals >= lo) & (score_vals < hi)
        else:
            msk = (score_vals >= lo) & (score_vals <= hi)
        pd_vals[msk] = float(cal_sub.loc[i, 'pd'])

    # Clip out-of-range
    msk_low = np.isnan(pd_vals) & (score_vals < min_edge)
    msk_high = np.isnan(pd_vals) & (score_vals > max_edge)
    if msk_low.any():
        pd_vals[msk_low] = float(cal_sub.iloc[0]['pd'])
    if msk_high.any():
        pd_vals[msk_high] = float(cal_sub.iloc[-1]['pd'])

    df_local['cal_pd'] = pd_vals
    cov = float(pd.Series(df_local['cal_pd']).notna().mean())
    if cov < float(min_coverage):
        raise ValueError('Fail-closed: cal_pd coverage=' + str(cov) + ' < min_coverage=' + str(min_coverage) + ' for model ' + model_key)

    return df_local


def build_policy_tables(scored_df_with_pd, prefer_pd=True, n_deciles=10):
    df_local = scored_df_with_pd.copy()

    if prefer_pd:
        if 'cal_pd' not in df_local.columns:
            raise ValueError('Fail-closed: cal_pd missing but prefer_pd=True')
        band_var = 'cal_pd'
        # lower PD is better; we want approve lowest PD
        sort_var = 'cal_pd'
        sort_ascending = True
    else:
        band_var = 'model_score'
        corr_val = float(pd.Series(df_local['model_score']).corr(pd.Series(df_local['y'])))
        if np.isnan(corr_val):
            raise ValueError('Fail-closed: correlation undefined in raw-score policy build')
        ascending_risk = bool(corr_val > 0)
        # best first ordering: lower risk first
        sort_var = 'model_score'
        sort_ascending = not ascending_risk

    df_sorted = df_local.sort_values(sort_var, ascending=sort_ascending).reset_index(drop=True)
    df_sorted['rank'] = np.arange(1, int(df_sorted.shape[0]) + 1)
    df_sorted['approval_rate'] = df_sorted['rank'] / float(df_sorted.shape[0])
    df_sorted['cum_bad_rate'] = df_sorted['y'].cumsum() / df_sorted['rank']

    grid = np.linspace(0.05, 0.95, 19)
    rows = []
    for ar in grid:
        k = int(np.floor(ar * float(df_sorted.shape[0])))
        if k < 1:
            continue
        sub_df = df_sorted.iloc[:k]
        cutoff = float(sub_df[sort_var].max())
        rows.append([float(ar), int(k), float(sub_df['y'].mean()), cutoff])

    thresh_tbl = pd.DataFrame(rows, columns=['approve_rate_target','n_approved','expected_bad_rate','cutoff'])
    thresh_tbl['cutoff_var'] = sort_var

    # Decile bands
    df_local = df_local.replace([np.inf, -np.inf], np.nan).dropna(subset=[band_var,'y']).copy()
    df_local['band_decile'] = pd.qcut(df_local[band_var], q=int(n_deciles), labels=False, duplicates='drop')

    dec_tbl = df_local.groupby('band_decile').agg(
        n=('y','size'),
        obs_bad_rate=('y','mean'),
        band_min=(band_var,'min'),
        band_max=(band_var,'max'),
        band_mean=(band_var,'mean')
    ).reset_index().sort_values('band_decile')
    dec_tbl['band_var'] = band_var

    return thresh_tbl, dec_tbl, df_sorted


def run_locked_policy_pipeline(xgb_df_val_scored, lgb_df_val_scored,
                              n_bins=50, min_n=5000, min_bads=50,
                              min_coverage=0.98, export_prefix=''):
    # Phase 1: build calibration maps
    xgb_map = build_pd_calibration_map(xgb_df_val_scored, 'xgb', n_bins=n_bins, min_n=min_n, min_bads=min_bads)
    lgb_map = build_pd_calibration_map(lgb_df_val_scored, 'lgb', n_bins=n_bins, min_n=min_n, min_bads=min_bads)

    pd_calibration_map_tbl = pd.concat([xgb_map, lgb_map], axis=0, ignore_index=True)

    # Phase 2: attach cal_pd + build policy tables
    xgb_with_pd = attach_cal_pd(xgb_df_val_scored, pd_calibration_map_tbl, 'xgb', min_coverage=min_coverage)
    lgb_with_pd = attach_cal_pd(lgb_df_val_scored, pd_calibration_map_tbl, 'lgb', min_coverage=min_coverage)

    xgb_thresh, xgb_band, xgb_sorted = build_policy_tables(xgb_with_pd, prefer_pd=True)
    lgb_thresh, lgb_band, lgb_sorted = build_policy_tables(lgb_with_pd, prefer_pd=True)

    # Export
    xgb_thresh_path = export_prefix + 'xgb_approval_policy_by_approval_rate.csv'
    xgb_band_path = export_prefix + 'xgb_policy_by_band_decile.csv'
    lgb_thresh_path = export_prefix + 'lgb_approval_policy_by_approval_rate.csv'
    lgb_band_path = export_prefix + 'lgb_policy_by_band_decile.csv'

    xgb_thresh.to_csv(xgb_thresh_path, index=False)
    xgb_band.to_csv(xgb_band_path, index=False)
    lgb_thresh.to_csv(lgb_thresh_path, index=False)
    lgb_band.to_csv(lgb_band_path, index=False)

    artifacts = {
        'pd_calibration_map_tbl': pd_calibration_map_tbl,
        'xgb_policy_threshold_tbl': xgb_thresh,
        'xgb_policy_band_tbl': xgb_band,
        'lgb_policy_threshold_tbl': lgb_thresh,
        'lgb_policy_band_tbl': lgb_band,
        'xgb_policy_df_sorted': xgb_sorted,
        'lgb_policy_df_sorted': lgb_sorted,
        'paths': {
            'xgb_thresh': xgb_thresh_path,
            'xgb_band': xgb_band_path,
            'lgb_thresh': lgb_thresh_path,
            'lgb_band': lgb_band_path
        }
    }

    return artifacts

def attach_cal_pd_from_map(scored_df, cal_map_df, model_name, score_col):
    # Interval-join score -> calibrated PD using the map’s score_min/score_max
    cal_model = cal_map_df[cal_map_df["model"] == model_name].copy()
    cal_model = cal_model.sort_values("score_min")

    work_df = scored_df.copy()
    work_df["_score"] = pd.to_numeric(work_df[score_col], errors="coerce")

    # Build bin edges and use pd.cut to assign bins
    bin_edges = sorted(set(cal_model["score_min"].tolist() + [float(cal_model["score_max"].max())]))
    work_df["_bin"] = pd.cut(work_df["_score"], bins=bin_edges, include_lowest=True, right=True)

    # Map each bin -> PD (using max PD in that bin)
    cal_model["_bin"] = pd.cut(cal_model["score_min"], bins=bin_edges, include_lowest=True, right=True)
    lookup = cal_model.groupby("_bin", dropna=False).agg(cal_pd=("pd","max")).reset_index()

    out_df = work_df.merge(lookup, on="_bin", how="left")
    out_df = out_df.drop(columns=["_score","_bin"])
    return out_df

def add_policy_flags(agent_df, policy_threshold_tbl, prefix):
    out_df = agent_df.copy()
    for ar in [0.10, 0.20, 0.50, 0.80]:
        idx = policy_threshold_tbl["approve_rate_target"].sub(ar).abs().idxmin()
        cutoff_val = float(policy_threshold_tbl.loc[idx, "cutoff"])
        out_df[prefix + "_approved_at_" + str(int(ar*100))] = (out_df["cal_pd"] <= cutoff_val).astype(int)
    return out_df
