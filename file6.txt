def prepare_pd_training_and_validation_data(
    df_pd_raw,
    df_pd_transformed,
    target_col,
    train_cutoff,
    id_cols,
    protected_cols,
    pd_feature_blacklist,
    forbidden_feature_patterns,
    date_cols,
    split_date_col="snapshot_dt",
    allowed_features=None,
    debug=False,
):

    # 0) Force alignment on common index to avoid drift
    common_index = df_pd_raw.index.intersection(df_pd_transformed.index)
    df_pd_raw = df_pd_raw.loc[common_index].copy()
    df_pd_transformed = df_pd_transformed.loc[common_index].copy()
    assert df_pd_raw.index.equals(df_pd_transformed.index), \
        "[FATAL] Raw and transformed snapshots misaligned even after alignment"
    
    # 1) Hard guards on split date integrity (split_date_col can live in raw OR transformed)
    assert (split_date_col in df_pd_raw.columns) or (split_date_col in df_pd_transformed.columns), \
        "[ERROR] split_date_col missing: " + str(split_date_col)
    
    # Prefer raw for splitting (since transformed is numeric-only by design)
    if split_date_col in df_pd_raw.columns:
        df_pd_raw[split_date_col] = pd.to_datetime(df_pd_raw[split_date_col], errors="coerce")
        split_series = df_pd_raw[split_date_col]
    else:
        df_pd_transformed[split_date_col] = pd.to_datetime(df_pd_transformed[split_date_col], errors="coerce")
        split_series = df_pd_transformed[split_date_col]
    
    assert split_series.notna().all(), "[ERROR] split_date_col has NaT: " + str(split_date_col)
    
    # 1b) Coerce optional date columns if present (do NOT require them in transformed)
    for c in date_cols:
        if (c not in df_pd_raw.columns) and (c not in df_pd_transformed.columns):
            if debug:
                print("[WARN] Date col missing (skipping): " + str(c))
            continue
    
        if c in df_pd_raw.columns:
            if not pd.api.types.is_datetime64_any_dtype(df_pd_raw[c]):
                df_pd_raw[c] = pd.to_datetime(df_pd_raw[c], errors="coerce")
    
        if c in df_pd_transformed.columns:
            if not pd.api.types.is_datetime64_any_dtype(df_pd_transformed[c]):
                df_pd_transformed[c] = pd.to_datetime(df_pd_transformed[c], errors="coerce")

        # Normalize patterns once
    forbidden_patterns_low = [str(p).lower() for p in forbidden_feature_patterns]
    blacklist_low = set([str(c).lower() for c in pd_feature_blacklist])
    id_cols_low = set([str(c).lower() for c in id_cols])
    protected_low = set([str(c).lower() for c in protected_cols])
    
    # 2) Candidate numeric features
    base_cols = allowed_features if allowed_features is not None else df_pd_transformed.columns
    candidate_features = []
    
    for c in base_cols:
        if c not in df_pd_transformed.columns:
            continue
    
        c_low = str(c).lower()
        if c_low in id_cols_low:
            continue
        if c_low in protected_low:
            continue
        if c_low in blacklist_low:
            continue
        if c == target_col:
            continue
        if c == split_date_col:
            continue
        if not pd.api.types.is_numeric_dtype(df_pd_transformed[c]):
            continue
    
        candidate_features.append(c)
    
    if debug:
        print("[INFO] Candidate numeric features: " + str(len(candidate_features)))
    
    # 3) Pattern-based leakage guard (case-insensitive)
    leakage_hits = []
    for c in candidate_features:
        c_low = c.lower()
        if any(p in c_low for p in forbidden_patterns_low):
            leakage_hits.append(c)
    
    assert not leakage_hits, (
        "[ERROR] Pattern-based leakage detected. "
        "Update PD_FEATURE_BLACKLIST explicitly:\n"
        + ", ".join(leakage_hits)
    )

    # 4) Time-based split: train vs validation
    if split_date_col in df_pd_raw.columns:
        split_series = pd.to_datetime(df_pd_raw[split_date_col], errors="coerce")
    else:
        split_series = pd.to_datetime(df_pd_transformed[split_date_col], errors="coerce")
    
    assert split_series.notna().all(), "[ERROR] split_date_col has NaT: " + str(split_date_col)
    
    train_mask = split_series <= train_cutoff
    val_mask = ~train_mask
    
    df_train_trans = df_pd_transformed.loc[train_mask].copy()
    df_train_raw = df_pd_raw.loc[train_mask].copy()
    df_val_trans = df_pd_transformed.loc[val_mask].copy()
    df_val_raw = df_pd_raw.loc[val_mask].copy()  
    assert "agent_msisdn" in df_train_raw.columns, "[ERROR] agent_msisdn missing in df_train_raw"
    assert "agent_msisdn" in df_val_raw.columns, "[ERROR] agent_msisdn missing in df_val_raw"

    # Great Check
    assert df_train_raw["agent_msisdn"].notna().all(), "[ERROR] agent_msisdn nulls in train"
    assert df_val_raw["agent_msisdn"].notna().all(), "[ERROR] agent_msisdn nulls in val"

    agent_train = df_train_raw["agent_msisdn"].copy()
    agent_val = df_val_raw["agent_msisdn"].copy()
    
    assert not df_train_raw.empty, "[ERROR] Training set is empty"
    assert df_train_raw.shape[0] == df_train_trans.shape[0], \
        "[ERROR] Raw and transformed train rows misaligned (row count)"
    assert df_train_raw.index.equals(df_train_trans.index), \
        "[ERROR] Raw and transformed train indexes misaligned"
    
    assert df_val_raw.shape[0] == df_val_trans.shape[0], \
        "[ERROR] Raw and transformed validation rows misaligned (row count)"
    assert df_val_raw.index.equals(df_val_trans.index), \
        "[ERROR] Raw and transformed validation indexes misaligned"

    # 5) Build X / y matrices
    assert target_col in df_train_trans.columns, "[ERROR] target_col missing in train transformed: " + str(target_col)
    assert target_col in df_val_trans.columns, "[ERROR] target_col missing in val transformed: " + str(target_col)
    
    missing_train_raw = [c for c in candidate_features if c not in df_train_raw.columns]
    missing_train_trans = [c for c in candidate_features if c not in df_train_trans.columns]
    assert not missing_train_raw, "[ERROR] Candidate features missing in train raw: " + str(missing_train_raw)
    assert not missing_train_trans, "[ERROR] Candidate features missing in train transformed: " + str(missing_train_trans)
    
    missing_val_raw = [c for c in candidate_features if c not in df_val_raw.columns]
    missing_val_trans = [c for c in candidate_features if c not in df_val_trans.columns]
    assert not missing_val_raw, "[ERROR] Candidate features missing in val raw: " + str(missing_val_raw)
    assert not missing_val_trans, "[ERROR] Candidate features missing in val transformed: " + str(missing_val_trans)

    assert "agent_msisdn" not in candidate_features, "[FATAL] agent_msisdn leaked into candidate_features"
    
    X_train_raw = df_train_raw[candidate_features]
    X_train_trans = df_train_trans[candidate_features]
    y_train = df_train_trans[target_col]

    thin_train = df_train_raw["thin_file_flag"]
    thin_val = df_val_raw["thin_file_flag"]
    
    X_val_raw = df_val_raw[candidate_features]
    X_val_trans = df_val_trans[candidate_features]
    y_val = df_val_trans[target_col]
    
    assert X_train_raw.shape == X_train_trans.shape, "[ERROR] Train raw/trans shapes differ"
    assert X_val_raw.shape == X_val_trans.shape, "[ERROR] Val raw/trans shapes differ"
    
    assert y_train.notna().all(), "[ERROR] Training target contains NaNs"
    assert y_val.notna().all(), "[ERROR] Validation target contains NaNs"
    
    if debug:
        print("[INFO] Training rows: " + str(len(X_train_raw)))
        print("[INFO] Validation rows: " + str(len(X_val_raw)))
        print("[INFO] Features passed to IV stage: " + str(len(candidate_features)))
    
   

    return (
    X_train_raw,
    X_train_trans,
    y_train,
    X_val_raw,
    X_val_trans,
    y_val,
    candidate_features,
    df_train_raw["thin_file_flag"],
    df_val_raw["thin_file_flag"],
    agent_train,
    agent_val,
    )

def iv_audit(
    iv_df,
    X_train_transformed,
    y_train,
    forbidden_patterns=FORBIDDEN_FEATURE_PATTERNS,
    blacklist=PD_FEATURE_BLACKLIST,
    high_iv_threshold=10.0,
    leak_corr_threshold=0.99,
):
    """
    Annotates an IV table for leakage, degenerate features, and extreme IV values.

    Args:
        iv_df: IV table from iv_filter_phase_2
        X_train_transformed: DataFrame used to compute IV
        y_train: PD target (1/0)
        forbidden_patterns: iterable of substrings marking forbidden columns
        blacklist: iterable of columns never allowed
        high_iv_threshold: IV above this value triggers review
        leak_corr_threshold: abs(corr) with target above this flags potential leakage

    Returns:
        iv_audit_df: IV table with 'status' and 'status_detail' columns
    """
  
       # --- basic validation to avoid silent weirdness later ---
    needed_cols = {"feature", "iv_after"}
    missing_cols = needed_cols - set(iv_df.columns)
    if len(missing_cols) > 0:
        raise ValueError("iv_df missing required columns: " + str(sorted(list(missing_cols))))

    blacklist_set = set(blacklist) if blacklist is not None else set()
    forbidden_patterns = tuple(forbidden_patterns) if forbidden_patterns is not None else tuple()

    # Align y to X index (prevents shuffled-row correlation bugs)
    y_series = pd.Series(y_train, index=X_train_transformed.index)

    iv_audit_records = []

    for _, row in iv_df.iterrows():
        feat = row["feature"]
        iv_after = row["iv_after"]
        iv_before = row["iv_before"] if "iv_before" in row.index else np.nan
        iv_uplift = row["iv_uplift"] if "iv_uplift" in row.index else np.nan

        status = "OK"
        status_detail = ""

        if feat not in X_train_transformed.columns:
            status = "SKIPPED_NOT_IN_X"
            status_detail = "Feature not found in X_train_transformed"
        else:
            col = X_train_transformed[feat]

            if feat in blacklist_set or any(p in feat for p in forbidden_patterns):
                status = "SKIPPED_FORBIDDEN"
                status_detail = "In PD_FEATURE_BLACKLIST or matches forbidden pattern"

            elif not pd.api.types.is_numeric_dtype(col):
                status = "SKIPPED_NON_NUMERIC"
                status_detail = "Non-numeric dtype"

            else:
                # Explicit missing / constant checks
                non_null_count = int(col.notna().sum())
                if non_null_count == 0:
                    status = "SKIPPED_ALL_MISSING"
                    status_detail = "All values are "
                elif col.nunique(dropna=True) <= 1:
                    status = "SKIPPED_CONSTANT"
                    status_detail = "Constant or single unique value"
                else:
                    # All zeros among non-missing
                    if (col.dropna() == 0).all():
                        status = "SKIPPED_CONSTANT"
                        status_detail = "All non-missing values are zero"

                    # High IV flag
                    if status == "OK" and (pd.notna(iv_after) and iv_after >= high_iv_threshold):
                        status = "REVIEW_HIGH_IV"
                        status_detail = "IV_after >= " + str(high_iv_threshold)

                    # Correlation with target (index-aligned)
                    corr = np.nan
                    try:
                        xy = pd.concat([col, y_series], axis=1).dropna()
                        if xy.shape[0] > 2:
                            corr = xy.iloc[:, 0].corr(xy.iloc[:, 1])
                    except Exception:
                        corr = np.nan

                    if pd.notna(corr) and abs(corr) >= leak_corr_threshold:
                        if status == "OK":
                            status = "REVIEW_LEAKAGE"
                        else:
                            status = status + "_AND_LEAKAGE"
                        if status_detail:
                            status_detail = status_detail + "; "
                        status_detail = status_detail + "abs(corr_with_target) >= " + str(leak_corr_threshold)

                    # Name-based review flags
                    leak_name_patterns = ("target", "label", "bad_state", "default", "dpd", "days_past_due")
                    id_patterns = ("id", "_id", "uuid", "guid", "account_id", "customer_id", "client_id", "agent_id")
                    date_patterns = ("date", "dt_", "_dt", "_timestamp")

                    lower_name = str(feat).lower()  
                    if any(p in lower_name for p in leak_name_patterns):
                        if status == "OK":
                            status = "REVIEW_LEAKAGE"
                        else:
                            status = status + "_AND_LEAKAGE"
                        if status_detail:
                            status_detail = status_detail + "; "
                        status_detail = status_detail + "Name suggests target/leakage"

                    if any(p in lower_name for p in id_patterns):
                        if status == "OK":
                            status = "REVIEW_ID"
                        else:
                            status = status + "_AND_ID"
                        if status_detail:
                            status_detail = status_detail + "; "
                        status_detail = status_detail + "Name suggests ID-like feature"

                    if any(p in lower_name for p in date_patterns):
                        if status == "OK":
                            status = "REVIEW_DATE"
                        else:
                            status = status + "_AND_DATE"
                        if status_detail:
                            status_detail = status_detail + "; "
                        status_detail = status_detail + "Name suggests time-like feature"

        iv_audit_records.append(
            {
                "feature": feat,
                "iv_before": iv_before,
                "iv_after": iv_after,
                "iv_uplift": iv_uplift,
                "status": status,
                "status_detail": status_detail,
            }
        )

    iv_audit_df = pd.DataFrame(iv_audit_records).sort_values("iv_after", ascending=False).reset_index(drop=True)
    return iv_audit_df

def resolve_feature_conflict(f1, f2, iv_map, monotonic_map, mono_iv_tolerance=0.05):
    """
    Decide which feature to KEEP when correlation is high.

    Policy:
    - If exactly one feature is monotonic-constrained, keep it unless it's worse by more than mono_iv_tolerance in IV.
    - Otherwise keep higher-IV.
    - Final tie-breaker is alphabetical for determinism.

    mono_iv_tolerance:
    Allow monotonic feature to be up to this much IV worse and still keep it.
    """

    def safe_iv(feat_name):
        iv_val = iv_map.get(feat_name, None)
        if iv_val is None or (isinstance(iv_val, float) and np.isnan(iv_val)):
            return float("-inf")
        return float(iv_val)

    def safe_mono(feat_name):
        mono_val = monotonic_map.get(feat_name, 0)
        try:
            mono_val = int(mono_val)
        except Exception:
            mono_val = 0
        return mono_val

    iv1 = safe_iv(f1)
    iv2 = safe_iv(f2)
    mono1 = safe_mono(f1)
    mono2 = safe_mono(f2)

    mono1_is = mono1 != 0
    mono2_is = mono2 != 0

    if mono1_is and not mono2_is:
        if iv1 + mono_iv_tolerance >= iv2:
            return f1
        return f2

    if mono2_is and not mono1_is:
        if iv2 + mono_iv_tolerance >= iv1:
            return f2
        return f1

    if iv1 > iv2:
        return f1
    if iv2 > iv1:
        return f2

    return f1 if f1 <= f2 else f2

def spearman_set_monotone_direction_safe(
    X_train,
    y_train,
    iv_table=None,
    monotonic_map=None,
    min_abs_target_spearman=0.05,
    override_nonzero_base=True
):
    """
    
      - Keep ALL features in X_train
      - Compute spearman(feature, target) to set direction:
          +1 if corr >= min_abs_target_spearman
          -1 if corr <= -min_abs_target_spearman
           0 otherwise (or if corr is )
      - If monotonic_map is provided, optionally keep its non-zero values (override_nonzero_base=True)

    Returns:
      final_monotone_map: dict(feature -> -1/0/+1) aligned to ALL columns in X_train
      spearman_target: pd.Series of raw spearman correlations (signed)
      spearman_log_df: DataFrame with feature, spearman, abs_spearman, iv_after, base_mono, final_mono
    """
    feat_list = list(X_train.columns)

    # Safe IV map (optional; used only for logging)
    iv_map = {}
    if iv_table is not None:
        iv_tbl = iv_table.copy()
        if ("feature" in iv_tbl.columns) and ("iv_after" in iv_tbl.columns):
            iv_tbl = iv_tbl.drop_duplicates(subset=["feature"], keep="last")
            iv_map = iv_tbl.set_index("feature")["iv_after"].to_dict()

    # Base monotone map (optional)
    base_map = monotonic_map if monotonic_map is not None else {}

    # Compute target spearman for each feature (signed)
    spearman_vals = {}
    y_ser = pd.Series(y_train)

    for col in feat_list:
        x_ser = pd.Series(X_train[col])
        cval = x_ser.corr(y_ser, method="spearman")
        spearman_vals[col] = cval

    spearman_target = pd.Series(spearman_vals).replace([np.inf, -np.inf], np.nan)

    # Convert to monotone directions
    spearman_suggested = {}
    for col in feat_list:
        cval = spearman_target.get(col)
        if pd.isna(cval) or abs(cval) < float(min_abs_target_spearman):
            spearman_suggested[col] = 0
        else:
            spearman_suggested[col] = 1 if cval > 0 else -1

    # Merge with base monotonic_map: keep base non-zero values if requested
    final_monotone_map = {}
    for col in feat_list:
        base_val = int(base_map.get(col, 0))
        if override_nonzero_base and base_val != 0:
            final_monotone_map[col] = base_val
        else:
            final_monotone_map[col] = int(spearman_suggested.get(col, 0))

    # Build log for debugging / transparency
    log_rows = []
    for col in feat_list:
        cval = spearman_target.get(col)
        abs_cval = np.nan if pd.isna(cval) else abs(float(cval))
        log_rows.append({
            "feature": col,
            "spearman_target": np.nan if pd.isna(cval) else float(cval),
            "abs_spearman_target": abs_cval,
            "iv_after": float(iv_map.get(col, np.nan)) if len(iv_map) else np.nan,
            "base_mono": int(base_map.get(col, 0)),
            "spearman_mono": int(spearman_suggested.get(col, 0)),
            "final_mono": int(final_monotone_map.get(col, 0)),
        })

    spearman_log_df = pd.DataFrame(log_rows).sort_values(
        "abs_spearman_target", ascending=False, na_position="last"
    )

    return final_monotone_map, spearman_target, spearman_log_df


def _psi_from_bins(expected, actual, eps=1e-6, assume_proportions_if_sum_close_to_1=True):
    expected = np.asarray(expected, dtype=float)
    actual = np.asarray(actual, dtype=float)

    if expected.shape != actual.shape:
        raise ValueError(
            "expected and actual must have same shape. Got "
            + str(expected.shape) + " and " + str(actual.shape)
        )

    # Replace NaNs/infs and enforce non-negative
    expected = np.nan_to_num(expected, nan=0.0, posinf=0.0, neginf=0.0)
    actual   = np.nan_to_num(actual,   nan=0.0, posinf=0.0, neginf=0.0)

    expected = np.maximum(expected, 0.0)
    actual   = np.maximum(actual, 0.0)

    exp_sum = float(expected.sum())
    act_sum = float(actual.sum())

    # Normalize counts to proportions (unless already ~1)
    if exp_sum > 0:
        if (not assume_proportions_if_sum_close_to_1) or (abs(exp_sum - 1.0) > 1e-3):
            expected = expected / exp_sum

    if act_sum > 0:
        if (not assume_proportions_if_sum_close_to_1) or (abs(act_sum - 1.0) > 1e-3):
            actual = actual / act_sum

    expected = np.clip(expected, eps, None)
    actual   = np.clip(actual, eps, None)

    return float(np.sum((actual - expected) * np.log(actual / expected)))
def compute_feature_psi(train_series: pd.Series, val_series: pd.Series, n_bins: int = 10, include_na_bin: bool = True):
    """
    PSI using quantile bins from TRAIN only.
    Robust to val values outside train range.
    Optionally includes NaNs as their own bin.
    """
   
    train_vals_raw = pd.to_numeric(train_series, errors="coerce")
    val_vals_raw = pd.to_numeric(val_series, errors="coerce")

    if include_na_bin:
        train_na = float(train_vals_raw.isna().mean())
        val_na = float(val_vals_raw.isna().mean())
        train_vals = train_vals_raw.dropna()
        val_vals = val_vals_raw.dropna()
    else:
        train_vals = train_vals_raw.dropna()
        val_vals = val_vals_raw.dropna()
        train_na = None
        val_na = None

    # If there are no non-missing values in train, PSI is purely missingness drift (if requested)
    if len(train_vals) == 0:
        if include_na_bin and train_na is not None and val_na is not None:
            return _psi_from_bins(np.array([1.0 - train_na, train_na]), np.array([1.0 - val_na, val_na]))
        return np.nan

    # If non-missing train is constant-ish, PSI is either 0 (no missing bin) or missingness drift only
    if train_vals.nunique(dropna=True) <= 1:
        if include_na_bin and train_na is not None and val_na is not None:
            return _psi_from_bins(np.array([1.0 - train_na, train_na]), np.array([1.0 - val_na, val_na]))
        return 0.0

    # Use effective bins based on uniqueness to reduce duplicate quantiles
    uniq_cnt = int(train_vals.nunique(dropna=True))
    n_bins_eff = int(min(n_bins, uniq_cnt))
    if n_bins_eff < 2:
        if include_na_bin and train_na is not None and val_na is not None:
            return _psi_from_bins(np.array([1.0 - train_na, train_na]), np.array([1.0 - val_na, val_na]))
        return 0.0

    try:
        q_grid = np.linspace(0, 1, n_bins_eff + 1)
        q_vals = np.quantile(train_vals.values, q=q_grid)
        bins_core = np.unique(q_vals)
    except Exception:
        return np.nan

    # If ties collapse bins too much, fall back to missingness-only PSI if applicable
    if len(bins_core) <= 2:
        if include_na_bin and train_na is not None and val_na is not None:
            return _psi_from_bins(np.array([1.0 - train_na, train_na]), np.array([1.0 - val_na, val_na]))
        return 0.0

    # Build histogram edges
    bins = np.concatenate(([-np.inf], bins_core[1:-1], [np.inf]))

    train_counts, _ = np.histogram(train_vals.values, bins=bins)
    val_counts, _ = np.histogram(val_vals.values, bins=bins)

    train_sum = float(train_counts.sum())
    val_sum = float(val_counts.sum())

    train_dist = train_counts / train_sum if train_sum > 0 else np.zeros_like(train_counts, dtype=float)
    val_dist = val_counts / val_sum if val_sum > 0 else np.zeros_like(val_counts, dtype=float)

    if include_na_bin and train_na is not None and val_na is not None:
        train_dist = np.append(train_dist * (1.0 - train_na), train_na)
        val_dist = np.append(val_dist * (1.0 - val_na), val_na)

    return _psi_from_bins(train_dist, val_dist)


def psi_validation_with_thin_file(
    X_train: pd.DataFrame,
    X_val: pd.DataFrame,
    thin_file_flag_col: str = "thin_file_flag",
    n_bins: int = 10,
    psi_drop_threshold: float = 0.25,
    psi_monitor_threshold: float = 0.10,
    nan_psi_decision: str = "MONITOR",
):
    """
    Computes PSI:
    - Overall
    - Thin-file only
    - Thick-file only

    Returns a PSI decision table
    """


    # Register psi_validation_with_thin_file (fix indentation) and make it robust to constant thin_file_flag

    # Update psi_validation_with_thin_file to never use float("") placeholders and to handle degenerate thin flag safely

    results = []

    # If thin flag missing in either dataset, fall back to overall PSI only
    has_flag = (thin_file_flag_col in X_train.columns) and (thin_file_flag_col in X_val.columns)
    if not has_flag:
        feature_cols = [c for c in X_train.columns if c in X_val.columns]
        for col in feature_cols:
            psi_overall = compute_feature_psi(X_train[col], X_val[col], n_bins)

            if pd.isna(psi_overall):
                decision = nan_psi_decision
            elif psi_overall > psi_drop_threshold:
                decision = "DROP"
            elif psi_overall >= psi_monitor_threshold:
                decision = "MONITOR"
            else:
                decision = "KEEP"

            results.append({
                "feature": col,
                "psi_overall": float(psi_overall) if pd.notna(psi_overall) else np.nan,
                "psi_thin_file": np.nan,
                "psi_thick_file": np.nan,
                "decision": decision
            })  
        psi_df = pd.DataFrame(results)
        decision_rank_map = {"DROP": 0, "MONITOR": 1, "KEEP": 2}
        psi_df["decision_rank"] = psi_df["decision"].map(decision_rank_map).fillna(99)
        psi_df = psi_df.sort_values(by=["decision_rank", "psi_overall"], ascending=[True, False]).drop(columns=["decision_rank"]).reset_index(drop=True)

        psi_df["psi_overall"] = psi_df["psi_overall"].round(4)
        return psi_df

    # Thin flag present: compute overall + subgroup PSIs when subgroup exists in BOTH splits
    feature_cols = [c for c in X_train.columns if (c != thin_file_flag_col and c in X_val.columns)]

    train_flag = pd.to_numeric(X_train[thin_file_flag_col], errors="coerce").fillna(0)
    val_flag = pd.to_numeric(X_val[thin_file_flag_col], errors="coerce").fillna(0)

    train_thin_mask = (train_flag == 1)
    val_thin_mask = (val_flag == 1)
    train_thick_mask = ~train_thin_mask
    val_thick_mask = ~val_thin_mask

    train_has_thin = bool(train_thin_mask.any())
    val_has_thin = bool(val_thin_mask.any())
    train_has_thick = bool(train_thick_mask.any())
    val_has_thick = bool(val_thick_mask.any())

    for col in feature_cols:
        psi_overall = compute_feature_psi(X_train[col], X_val[col], n_bins)

        psi_thin = np.nan
        if train_has_thin and val_has_thin:
            psi_thin = compute_feature_psi(X_train.loc[train_thin_mask, col], X_val.loc[val_thin_mask, col], n_bins)

        psi_thick = np.nan
        if train_has_thick and val_has_thick:
            psi_thick = compute_feature_psi(X_train.loc[train_thick_mask, col], X_val.loc[val_thick_mask, col], n_bins)

        psi_vals_for_drop = []
        if pd.notna(psi_overall):
            psi_vals_for_drop.append(float(psi_overall))
        if pd.notna(psi_thin):
            psi_vals_for_drop.append(float(psi_thin))
        if pd.notna(psi_thick):
            psi_vals_for_drop.append(float(psi_thick))

        if len(psi_vals_for_drop) == 0:
            decision = nan_psi_decision
        elif max(psi_vals_for_drop) > psi_drop_threshold:
            decision = "DROP"
        elif (pd.notna(psi_overall) and psi_overall >= psi_monitor_threshold) or (pd.notna(psi_thin) and psi_thin >= psi_monitor_threshold) or (pd.notna(psi_thick) and psi_thick >= psi_monitor_threshold):
            decision = "MONITOR"
        else:
            decision = "KEEP"

        results.append({
            "feature": col,
            "psi_overall": float(psi_overall) if pd.notna(psi_overall) else np.nan,
            "psi_thin_file": float(psi_thin) if pd.notna(psi_thin) else np.nan,
            "psi_thick_file": float(psi_thick) if pd.notna(psi_thick) else np.nan,
            "decision": decision
        })

    psi_df = pd.DataFrame(results)
    decision_rank_map = {"DROP": 0, "MONITOR": 1, "KEEP": 2}
    psi_df["decision_rank"] = psi_df["decision"].map(decision_rank_map).fillna(99)
    psi_df = psi_df.sort_values(by=["decision_rank", "psi_overall", "psi_thin_file"], ascending=[True, False, False]).drop(columns=["decision_rank"]).reset_index(drop=True)

    psi_df["psi_overall"] = psi_df["psi_overall"].round(4)
    psi_df["psi_thin_file"] = psi_df["psi_thin_file"].round(4)
    psi_df["psi_thick_file"] = psi_df["psi_thick_file"].round(4)

    return psi_df
