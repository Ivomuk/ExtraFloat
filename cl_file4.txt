def map_hdb_to_tier(cluster_id):
    if cluster_id == -1:
        return "Noise / Irregular"
    # Very top band: highest activity & value
    if cluster_id in [3]:
        return "Platinum Power"
    # Strong but slightly below Platinum
    if cluster_id in [5]:
        return "Gold Power"
    # Solid mid-high performers
    if cluster_id in [6]:
        return "Silver Strong"
    # Mid-tier active
    if cluster_id in [7]:
        return "Bronze Active"
    # Lower-mid active bands along the ladder
    if cluster_id in [8, 10, 19]:
        return "Developing Active"
    # Very low-activity / semi-dormant band
    if cluster_id in [0]:
        return "Emerging / Low Activity"
    # Everything else: small dense pockets
    return "Other HDB Segment"
